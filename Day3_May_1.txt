function unique(str)
{
    let str1=[]
    str1=str[0]
    for(let i=1;i<str.length;i++){
        c=0
        for(let j=0;j<str1.length;j++)
        {
            if(str[i]==str1[j])
            {
                c++
                break;
            }
        }
        if(c==0)
        str1+=str[i]
    }
    return str1;
}
function factorial(n){
    fact=1
    for(i=1;i<=n;i++){
        fact*=i
    }
    return fact;
}
