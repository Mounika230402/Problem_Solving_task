1) Program to print 1 to num, num to 1, -1 to -num, -num to -1 by using while loop
-------------------------------------------------------------------------------------------------------------------
const prompt=require("prompt-sync")();


num= +prompt("enter a number: ");
console.log("1 to num");
let i=1
while(i!=num+1){
    console.log(i);
    i++;
}

console.log("num to 1");
let j=num
while(j!=0){
    console.log(j);
    j--;
}
console.log("-1 to -num");
let k=-1
while(k!=-num-1){
    console.log(k);
    k--;
}
console.log("-num to -1");
let m=-num
while(m!=0){
    console.log(m);
    m++;
}


2) Write a program to find the largest prime in the given number
input: 46537
output: 7

num = "46537";
max = 0;
for (n of num) {
    if (n > 1) {
        c = true;
        for (i = 2; i < n; i++) {
            if (n % i == 0) {
                c = false;
                break;
            }
        }
        if (c) {
            if (max < n)
                max = n;
        }
    }
}
console.log(max);
